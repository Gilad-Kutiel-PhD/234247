\begin{claim}
אם אין בגרף מעגלים שלילים אז קיים מסלול פשוט קל ביותר
\end{claim}
\begin{proof}
נסתכל על המסלול הקל ביותר עם הכי מעט מעגלים, נוריד מעגל אחד.
\end{proof}



\begin{claim}
אם
$p = (v_0, \ldots, v_k)$
מסלול קל ביותר מ-%
$v_0$
ל-%
$v_k$
אז לכל
$0 \leq i \leq j \leq k$
מתקיים ש-%
$(v_i, \ldots, v_j)$
מסלול קל ביותר בין 
$v_i$
ל-%
$v_j$.
\end{claim}
\begin{proof}
אם לא, נחליף את המסלול הקל ביותר בתת מסלול הקיים ונקבל מסלול קל יותר.
\end{proof}

\begin{claim}[אי שוויון המשולש]
לכל 
$u,v \in V, uv \in E$
מתקיים ש-%
$\delta(s,v) \leq \delta(s,u) + w(uv)$
\end{claim}
\begin{proof}
משקל המסלול הקל ביותר מ-$s$ ל-$u$ ומשם ל-$v$ הוא 
$\delta(s,u) + w(uv)$
(אבל יתכנו מסלולים קלים יותר ממסלול זה).
\end{proof}
